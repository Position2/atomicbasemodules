 <div class="slider">
  <div><img src="https://picsum.photos/500/500?image=1051"></div>
  <div><img src="https://picsum.photos/500/500?image=1057"></div>
  <div><img src="https://picsum.photos/500/500?image=1073"></div>
 </div>






  .slider {
    position: relative;
  }
  .slider > div {
    position: absolute;
  }






$(".slider > div:gt(0)").hide();
setInterval(function() {
  $('.slider > div:first-child')
    .fadeOut(1000)
    .next()
    .fadeIn(1000)
    .end()
    .appendTo('.slider');
}, 3000);
