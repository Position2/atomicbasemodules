$(document).ready(function() {
   var utm_source = Cookies.get('utm_source');
   var utm_content = Cookies.get('utm_content');
   var utm_campaign = Cookies.get('utm_campaign');
   var utm_medium = Cookies.get('utm_medium');
   var utm_term = Cookies.get('utm_term');
   var xuid = Cookies.get('xuid');

   $("#utm_source_hidden_checkout_field").val(utm_source);
   $("#utm_content_hidden_checkout_field").val(utm_content);
   $("#utm_campaign_hidden_checkout_field").val(utm_campaign);
   $("#utm_medium_hidden_checkout_field").val(utm_medium);
   $("#utm_term_hidden_checkout_field").val(utm_term);
   $("#xuid_hidden_checkout_field").val(xuid);
});
