 var slider = $('#slider-thumb .filter-by-section-description-wrapper').bxSlider({
      "mode": "fade",
      "auto": true,
      "pager": false,
      "speed": 1000,
      "pause": 9000,
      "controls": false,
      "onSlideBefore": function($slideElement, oldIndex, newIndex) {
        $("#slider-thumb .filter-by-section .dTab li[data-slide-index='" + newIndex + "']").addClass("active").siblings("li").removeClass("active");
      }
    });

$("#slider-thumb .filter-by-section .dTab li").click(function() {
      slider.goToSlide($(this).attr("data-slide-index"));
      $(this).addClass("active").siblings("li").removeClass("active");
      slider.stopAuto();
    });
