    var alphachar = String.fromCharCode(i),
      charAppered = $(".alpha-filter").filter(function() {
        return $(this).text().trim().charAt(0).toLowerCase() == alphachar.toLowerCase()
      }).get();
    $("
  • ", {
      "html": alphachar + "" + charAppered.length + "",
      "data-filter": alphachar.toLowerCase(),
      "class": (charAppered.length <= 0) ? "disabled" : ""
    }).appendTo(filterUl);
  }
  filterUl.appendTo($(".filtersection"));
  $("body").on("click", ".filtersection ul li", function() {
    var dis = $(this),
      disFilter = dis.attr("data-filter") || "";
    if (!dis.hasClass("disabled")) {
      dis.addClass("current").siblings("li").removeClass("current");
      if (disFilter != "" && disFilter != "*")
        $(".alpha-filter").hide().filter(function() {
          return $(this).text().trim().charAt(0).toLowerCase() == disFilter.toLowerCase()
        }).show();
      else
        $(".alpha-filter").show();
    }
  });
