function initPagination(itemSelector) {
  var perPage = 3;
  $("#content .each-content").removeClass(function(index, css) {
    return (css.match(/(^|\s)page-\S+/g) || []).join(' ');
  });
  $(itemSelector).addClass(function(key, val) {
    return "page-" + (Math.floor(key / perPage) + 1);
  })
  $("#paging")
    .pagination('destroy')
    .pagination({           
      items: $(itemSelector).length,
      itemsOnPage: perPage,
      cssStyle: 'light-theme',
      onPageClick: function(pgno, e) {
        e.preventDefault();
        $(itemSelector).hide(0).filter(".page-" + pgno).show(0)
      },
      onInit: function() {
        $(itemSelector).hide(0).filter(".page-1").show(0)
      }
  });
}

$(".smooth-scroll").click(function() {
  var dis = $(this),
      target = dis.data("href"),
      offset = parseInt($(target).offset().top),
      header = $(".menu-wrapper");
  dis.addClass("active").siblings().removeClass("active");
  var newOffset = ($(window).width() > 1024) ? offset - (header.outerHeight()) : offset;
  $('html,body').stop().animate({ scrollTop: newOffset }, 200);
});

function findActiveSection() {
  var winHO = $(window).height(),
      winH = winHO - 200;
  return $(".section").map(function() {
    var dis = $(this),
        disY = dis[0].getBoundingClientRect().top;
    if ((disY < winH) && (disY > -200))
      return dis;
    else if ((disY < winHO) && (disY > -200))
      return dis;
  });
}
var prop_scrol_time = "";
if ($(window).width() > 1024) {
  $(window).scroll(function() {
    clearTimeout(prop_scrol_time);
    prop_scrol_time = setTimeout(function() {
      var cActive = findActiveSection();
      $(".smooth-scroll").removeClass("active").filter("[data-href='#" + $(cActive[0]).attr("id") + "']").addClass("active");
    }, 200);
  })
}

(function($) {
  $(document).ready(function() {
    function isDevice() {
      return /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent);
    }
    function isDesktop() { return ($(window).width() > 1200) }
    function isMobile() { return ($(window).width() < 768) }
    function isTablet() { return ($(window).width() >= 768 && $(window).width() <= 1024) }
    function isOddRes() { return ($(window).width() > 1024 && $(window).width() <= 1200) }
    function isIpadPro() { return isDevice() && ($(window).width() >= 1366) }
    function isRealDevice() {
      return /Android|webOS|iPhone|iPad|macintosh|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent) && (typeof(window.orientation) != "undefined" || (navigator.maxTouchPoints > 1));
    }
    function initMview() {
      $("body").addClass("mobileview",isDevice());
    }
    function init() { 
      initMview();
    }
    init();
    var prop_resize_time = "",
        wRevent = isRealDevice() ? "orientationchange" : "resize";
    $(window).on(wRevent,function() {
      clearTimeout(prop_resize_time);
      prop_resize_time = setTimeout(init,200); // Use function which you init  
    });
  })
})(jQuery);
